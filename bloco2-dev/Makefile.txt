CC = gcc
CFLAGS = -Wall -g

soma: soma.o
	$(CC) soma.o -o soma

soma.o: soma.c
	$(CC) $(CFLAGS) -c soma.c -o soma.o

clean:
	rm -f soma soma.o soma.i soma.s

.PHONY: clean
